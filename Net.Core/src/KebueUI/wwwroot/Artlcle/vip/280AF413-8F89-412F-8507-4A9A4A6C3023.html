Winform tabControl选项页重写

                        
                            
     
 private void tabControlMain_DrawItem(object sender, DrawItemEventArgs e)
        {
            tabControlMain.Font = new Font("Calibri", 14);
            SolidBrush BackBrush = new SolidBrush(Color.Blue);//标签背景填充颜色
            Brush selected_color = Brushes.DimGray; //选中的项的背景色
            SolidBrush FrontBrush = new SolidBrush(Color.White); //标签文字填充颜色
            StringFormat StringF = new StringFormat();
            StringF.Alignment = StringAlignment.Center; //设置文字对齐方式
            StringF.LineAlignment = StringAlignment.Center;
            Rectangle Rec = tabControlMain.GetTabRect(e.Index); //获取标签头工作区域
            string title = tabControlMain.TabPages[e.Index].Text;
            if (e.Index == this.tabControlMain.SelectedIndex)    //当前选中的Tab页，设置不同的样式以示选中
            {
                e.Graphics.FillRectangle(selected_color, Rec); //改变选项卡标签的背景色 
                e.Graphics.DrawString(title, new Font("Calibri",14), FrontBrush, Rec, StringF);//PointF选项卡标题的位置 
                // Rec.Offset(Rec.Width - iconWidth - 3, 2);
            }
            else//非选中的
            {
                e.Graphics.FillRectangle(BackBrush, Rec);//绘制标签头背景颜色
                e.Graphics.DrawString(title, new Font("Calibri", 12), FrontBrush, Rec, StringF);//绘制标签头文字
            }
        }
